# Session Setup ,initialize message history in session state 
# instead of using a simple list to store the messages history, we use session state to persist the conversation history across user interactions. This way, when the user sends a new message, we can append it to the existing history and display the entire conversation thread in the chat interface.
if "message_history" not in st.session_state:
    st.session_state['message_history'] = []

# loading the massages/conversation history from the session state and displaying them in the chat interface. We iterate through the message history and use st.chat_message to display each message according to its role (user or assistant). This allows us to maintain a coherent conversation thread in the chat interface, showing both user inputs and assistant responses in the correct order.
for message in st.session_state['message_history']:
    with st.chat_message(message['role']):
        st.text(message['content'])

# taking user input from the chat input box and processing it. When the user types a message and submits it
chat_box=st.chat_input('Type Here')

if chat_box:
    # when the user submits a message, we append it to the message history in the session state and display it in the chat interface. 
    st.session_state['message_history'].append({'role': 'user' , 'content' : chat_box})
    with st.chat_message('user'):
        st.text(chat_box)

    # we then invoke the workflow with the user's message and stream the assistant's response back to the chat interface.
    # response = workflow.invoke({'message' : [HumanMessage(content=chat_box)]},config=config)
    # ai_message = response['message'][-1].content

    # not only showing the response from the llm also streaming the response to the chat interface to get a user friendly experience, instead of workflow.invoke(user message), we use workflow.stream(user message) to stream the response. this allow us to display ai assistant response in real time.
    with st.chat_message('assistant'):
        ai_message = st.write_stream(
            message_chunk.content for message_chunk,metadata in workflow.stream(
                {'message': [HumanMessage(content = chat_box)]},
                config =({'configurable' : {'thread_id' : 'thread-1'}}),
                stream_mode= 'messages'
            )
        )
    st.session_state['message_history'].append({'role': 'assistant' , 'content' :ai_message})
